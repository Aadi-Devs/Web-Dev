cd             # Shows current directory (same as pwd)
cd Documents   # Changes to Documents folder
dir            # Lists files and folders
mkdir newFolder    # Creates a new folder called newFolder
type nul > file.txt   # Creates an empty file.txt
type file.txt             # Shows content of file.txt
notepad file.txt          # Opens file.txt in Notepad for editing
move oldname.txt newname.txt   # Renames oldname.txt to newname.txt
move file.txt D:\NewFolder\    # Moves file.txt to another folder
copy file.txt copyfile.txt     # Copies file.txt to copyfile.txt
